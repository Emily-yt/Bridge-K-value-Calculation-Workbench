### 3.2 K值计算服务

#### 类定义

```javascript
class CalculationService {
  /**
   * 执行K值计算
   * @param {string} bridgeId - 桥梁ID
   * @param {number} spanIndex - 孔索引
   * @param {Object} input - 计算输入参数
   * @returns {Object} 计算结果
   */
  async calculate(bridgeId, spanIndex, input) {
    // 1. 获取桥梁信息
    const bridge = await this.getBridge(bridgeId);
    
    // 2. 获取固定参数
    const fixedParams = await this.getFixedParams(bridge.spanType);
    
    // 3. 查表获取偏载系数
    const eta = this.lookupEta(input, fixedParams);
    
    // 4. 计算主力组合弯矩
    const momentM = this.calculateMoment(input, fixedParams, eta);
    
    // 5. 计算分项K值
    const k1 = this.calculateK1(input, fixedParams, momentM);
    const k2 = this.calculateK2(input, fixedParams, momentM);
    const k3 = this.calculateK3(input, fixedParams, momentM);
    const k4 = this.calculateK4(input, fixedParams, momentM);
    
    // 6. 确定最终K值
    const kFinal = Math.min(k1, k2, k3, k4);
    
    return {
      input,
      intermediate: { etaM: eta, momentM, fixedParams },
      output: { k1, k2, k3, k4, kFinal }
    };
  }
  
  /**
   * 查表获取偏载系数（支持线性插值）
   */
  lookupEta(input, fixedParams) {
    const { beamPosition, curveRadius, eccentricityE } = input;
    const table = fixedParams.eccentricityTable.midspan;
    
    // 根据梁型选择对应表格
    let etaTable;
    if (beamPosition === '直线梁') {
      etaTable = table['直线梁'];
    } else {
      const rKey = `R${curveRadius}`;
      const position = beamPosition === '曲线外梁' ? '曲线外梁' : '曲线内梁';
      etaTable = table[position][rKey];
    }
    
    // 线性插值
    return this.interpolate(etaTable, eccentricityE);
  }
  
  /**
   * 计算主力组合弯矩
   */
  calculateMoment(input, fixedParams, eta) {
    const { ballastThicknessT, impactFactor } = input;
    const { mq0, mq1, mq2, mq3Per10cm, mL } = fixedParams.internalForces.midspan;
    
    // 根据梁型获取mq1
    const mq1Value = mq1[input.beamPosition];
    
    // M = M_q0 + M_q1 + M_q2 + (t/10)*M_q3 + (1+μ)*η*M_L
    const moment = mq0 + mq1Value + mq2 + 
                   (ballastThicknessT / 10) * mq3Per10cm + 
                   impactFactor * eta * mL['中-活载'];
    
    return moment;
  }
  
  /**
   * 计算K1（正截面抗弯强度）
   */
  calculateK1(input, fixedParams, momentM) {
    const { damageFactors, beamPosition, impactFactor } = input;
    const { z1m } = damageFactors;
    const { mp, safetyFactor } = fixedParams;
    const { mq0, mq1, mq2, mq3Per10cm, mL } = fixedParams.internalForces.midspan;
    
    // 根据梁型获取mp和mq1
    const mpValue = beamPosition === '直线梁' ? mp['直线梁'] : mp['曲线梁'];
    const mq1Value = mq1[beamPosition];
    
    // 分子: Z1M * Mp / K^w - (M_q0 + M_q1 + M_q2 + t/10*M_q3)
    const numerator = z1m * mpValue / safetyFactor - 
                      (mq0 + mq1Value + mq2 + input.ballastThicknessT / 10 * mq3Per10cm);
    
    // 分母: (1+μ) * η * M_L
    const eta = this.lookupEta(input, fixedParams);
    const denominator = impactFactor * eta * mL['中-活载'];
    
    return numerator / denominator;
  }
  
  /**
   * 线性插值
   */
  interpolate(table, value) {
    const keys = Object.keys(table).map(Number).sort((a, b) => a - b);
    
    // 边界处理
    if (value <= keys[0]) return table[keys[0]];
    if (value >= keys[keys.length - 1]) return table[keys[keys.length - 1]];
    
    // 找到相邻两点
    let lower = keys[0];
    let upper = keys[keys.length - 1];
    for (let i = 0; i < keys.length - 1; i++) {
      if (value >= keys[i] && value <= keys[i + 1]) {
        lower = keys[i];
        upper = keys[i + 1];
        break;
      }
    }
    
    // 线性插值
    const t = (value - lower) / (upper - lower);
    return table[lower] + t * (table[upper] - table[lower]);
  }
}
```

## 五、业务逻辑层

### 5.1 计算流程

```
┌─────────────────────────────────────────────────────────────────┐
│                        K值计算业务流程                           │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  1. 接收计算请求                                                │
│     └─→ 校验输入参数（必填项、范围、类型）                       │
│         └─→ 校验失败：返回错误信息                               │
│                                                                 │
│  2. 获取桥梁信息                                                │
│     └─→ 查询bridges.json                                        │
│         └─→ 不存在：返回错误                                     │
│                                                                 │
│  3. 获取固定参数                                                │
│     └─→ 根据spanType查询fixed_params.json                       │
│         └─→ 缓存中读取或文件读取                                │
│                                                                 │
│  4. 查表获取偏载系数                                            │
│     └─→ 根据e、R、梁型查表                                      │
│         └─→ 需要时进行线性插值                                  │
│                                                                 │
│  5. 计算主力组合弯矩                                            │
│     └─→ M = M_q0 + M_q1 + M_q2 + t/10*M_q3 + (1+μ)*η*M_L       │
│                                                                 │
│  6. 计算分项K值                                                 │
│     ├─→ K1: 正截面抗弯强度                                      │
│     ├─→ K2: 正截面抗裂性                                        │
│     ├─→ K3: 正截面应力                                          │
│     └─→ K4: 斜截面应力                                          │
│                                                                 │
│  7. 确定最终K值                                                 │
│     └─→ K = min{K1, K2, K3, K4}                                │
│                                                                 │
│  8. 保存计算记录                                                │
│     └─→ 写入k_calculations.json                                 │
│         └─→ 生成唯一ID、记录时间戳                              │
│                                                                 │
│  9. 返回计算结果                                                │
│     └─→ 包含输入、中间结果、输出、状态                          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```
